import B;

class ObjectAssignmentFail {

    int bigBoy = 3;

    public static void main(String[] args) {
        B.println(bigBoy);
    }

    static public ObjectAssignmentFail foo() {
        ObjectAssignmentFail a;
        B b;

        a = new ObjectAssignmentFail();
        b = new B();

        // ObjectAssignmentFail does not extend B
        b = a;

        return a;
    }
}
